       let current = url;
       for (let hop = 0; hop <= maxHops; hop++) {
         let parsed;
         try { parsed = new URL(current); } catch { throw new Error(`Invalid URL: ${current}`); }
         if (parsed.protocol !== 'http:' && parsed.protocol !== 'https:') throw new Error(`Unsupported URL protocol: ${parsed.protocol}`);
         if (isPrivateHostname(parsed.hostname)) throw new Error(`URL points to a private/internal address: ${parsed.hostname}`);
         const res = await fetch(current, { redirect: 'manual', signal });
         if (res.status >= 300 && res.status < 400) {
           const loc = res.headers.get('location');
           if (!loc) throw new Error(`Redirect without Location header (HTTP ${res.status})`);
           current = new URL(loc, current).toString();

         }

       }
